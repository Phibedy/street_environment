1% init: [scalar] set to "1" once when object is first detected to initialize state vector, then set to "0"
 2% laneModel: [px1] state vector of the current lane model
 3% lanePieceLength: [scalar] length of linear pieces in the lane model
 4% r: [3x1] state vector of the object being tracked
 5%     [ s (arc length from vehicle to object in the diection of the lane model); 
 6%       v (velocity of object); 
 7%       a (acceleration of object)]
 8% Pk:[3x3] covariance of the current state 
 9%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% old: Q: [3x3] covariance of the state transition
10% Q: [scalar] standard deviation of the state transition in x-direction (how reliable is delta_x?) 
11% R: [scalar] variance of the measurement (e.g. the measured distance)
12% x_measure: [nx1] measured x values of the object being tracked
13% y_measure: [nx1] measured y values of the object being tracked
14% delta_x: [scalar] forward movement of the vehicle relative to the lane model
15
16function [r, Pk] = objectTracker(init, laneModel, lanePieceLength, r, Pk, Q, R, x_measure, y_measure, delta_x)
17
18
19%% find distance between lane model and object measurements to get the measurment of s (s_measure)
20[xp, yp, phi] = getPointsFromState(laneModel, lanePieceLength); 
21P = [xp, yp, phi];
22D = 10000*ones(numel(x_measure), 3); 
23
24for s=1:numel(laneModel)
25   for m=1:numel(x_measure)            
26        dist_point = sqrt((P(s, 1)-(x_measure(m)))^2 + (P(s, 2)-(y_measure(m)))^2);
27        if dist_point < D(m, 3)
28            D(m, 1) = s;
29            D(m, 2) = 0;
30            D(m, 3) = dist_point;            
31        end            
32        if s > 1
33            [dist_line, lambda, ~] = d_line_point([P(s-1, 1), P(s-1, 2)], [P(s, 1), P(s, 2)], [x_measure(m), y_measure(m)]);
34            if dist_line < D(m, 3) && lambda > 0 && lambda < 1
35                D(m, 1) = s-1;
36                D(m, 2) = lambda;
37                D(m, 3) = dist_line;              
38            end
39        end         
40    end 
41end
42
43s_measure = (mean(D(:,1) + D(:,2))-1)*lanePieceLength;
44
45% initialize the state vector on first object detection
46if init == 1
47    r = [s_measure; 0; 0];
48    return
49end
50
51
52%% state space model matrices
53T = 0.01;
54A =    [1, T, 0.5*T^2;
55        0, 1, T;
56        0, 0, 1];    
57B = [-1; 0; 0];    
58H = [1, 0, 0];
59
60% covariance of the state transition
61d = Q^2;
62QQ = [  d,     d*T,   d*T^2;
63        d*T,   d*T^2, d*T^3;
64        d*T^2, d*T^3, d*T^4];
65
66%% Kalman filter 
67
68% prediction
69r = A*r + B*delta_x;
70Pk = A*Pk*A' + QQ;
71
72% kalman gain
73S = H*Pk*H' + R;
74K = Pk*H'/S;
75
76% update
77r = r + K*(s_measure - H*r);
78Pk = Pk - K*S*K';
79
