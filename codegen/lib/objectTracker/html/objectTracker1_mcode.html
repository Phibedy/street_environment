1% init: [scalar] set to "1" once when object is first detected to initialize state vector, then set to "0"
 2% laneModel: [px1] state vector of the current lane model
 3% lanePieceLength: [scalar] length of linear pieces in the lane model
 4% r: [4x1] state vector of the object being tracked
 5%     [ s_x (arc length from vehicle to object in the diection of the lane model); 
 6%       v_x (velocity of object in direction of lane); 
 7%       s_y (lateral position of object relative to middle lane);
 8%       v_y (lateral velocity of object)]
 9% Pk:[4x4] covariance of the current state 
10% Q: [scalar] standard deviation of the state transition (how reliable is delta_x?) 
11% Rx: [scalar] standard deviation of the measurement in lane direction 
12% Ry: [scalar] standard deviation of the measurement lateral to the lane 
13% x_measure: [nx1] measured x values of the object being tracked
14% y_measure: [nx1] measured y values of the object being tracked
15% delta_x: [scalar] forward movement of the vehicle relative to the lane model
16
17function [r, Pk] = objectTracker(init, laneModel, lanePieceLength, r, Pk, Q, Rx, Ry, x_measure, y_measure, delta_x, hasMeasurement)
18% x und y Werte sind in Fzg-Koordinaten
19
20%% find distance between lane model and object measurements to get the measurment of s (s_measure)
21[xp, yp, phi] = getPointsFromState(laneModel, lanePieceLength); 
22P = [xp, yp, phi];
23D = 10000*[1 1 1 1];
24
25for s=1:numel(laneModel)         
26        dist_point = sqrt((P(s, 1)-(x_measure))^2 + (P(s, 2)-(y_measure))^2);
27        if dist_point < D(3)
28            D(1) = s;
29            D(2) = 0;
30            D(3) = dist_point;     
31            D(4) = -sign(laneModel(min([numel(laneModel), s+1])));  % positive Kr√ºmmung -> vorzeichen von d negativ (und anders rum)     
32        end            
33        if s > 1
34            [dist_line, lambda, sgn] = d_line_point([P(s-1, 1), P(s-1, 2)], [P(s, 1), P(s, 2)], [x_measure, y_measure]);
35            if dist_line < D(3) && lambda > 0 && lambda < 1
36                D(1) = s-1;
37                D(2) = lambda;
38                D(3) = dist_line; 
39                D(4) = sgn;  % vorzeichen von d
40            end
41        end     
42end
43
44y_measure = D(4)*D(3);
45x_measure = (D(1) + D(2)-1)*lanePieceLength;
46
47% initialize the state vector on first object detection
48if init == 1
49    r = [x_measure; 0; y_measure; 0];
50    return
51end
52
53
54%% state space model matrices
55T = 0.01;
56A =    [1, T, 0, 0;
57        0, 1, 0, 0;
58        0, 0, 1, T;
59        0, 0, 0, 1];    
60B = [-1; 0; 0; 0];    
61H = [1, 0, 0, 0;
62     0, 0, 1, 0];
63
64% covariance of the state transition
65d = Q^2;
66QQ = [  d,     d*T,   0,     0;
67        d*T,   d*T^2, 0,     0;
68        0,     0,     d,     d*T;
69        0,     0,     d*T,   d*T^2];
70
71%% Kalman filter 
72
73% prediction
74r = A*r + B*delta_x;
75Pk = A*Pk*A' + QQ;
76
77% kalman gain
78S = H*Pk*H' + [Rx^2, 0; 0 Ry^2];
79K = Pk*H'/S;
80
81% update
82if hasMeasurement
83    r = r + K*([x_measure; y_measure] - H*r);    
84end
85
86Pk = Pk - K*S*K';
87
