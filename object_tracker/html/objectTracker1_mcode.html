1% init: [scalar] set to "1" once when object is first detected to initialize state vector, then set to "0"
 2% laneModel: [px1] state vector of the current lane model
 3% lanePieceLength: [scalar] length of linear pieces in the lane model
 4% r: [3x1] state vector of the object being tracked
 5%     [ s_x (arc length from vehicle to object in the diection of the lane model); 
 6%       v_x (velocity of object in direction of lane); 
 7%       s_y (lateral position of object relative to middle lane);
 8%       v_y (lateral velocity of object)]
 9% Pk:[3x3] covariance of the current state 
10% Q: [scalar] standard deviation of the state transition (how reliable is delta_x?) 
11% R: [scalar] variance of the measurement (e.g. the measured distance)
12% x_measure: [nx1] measured x values of the object being tracked
13% y_measure: [nx1] measured y values of the object being tracked
14% delta_x: [scalar] forward movement of the vehicle relative to the lane model
15
16function [r, Pk] = objectTracker(init, laneModel, lanePieceLength, r, Pk, Q, Rx, Ry, x_measure, y_measure, delta_x)
17
18
19%% find distance between lane model and object measurements to get the measurment of s (s_measure)
20[xp, yp, phi] = getPointsFromState(laneModel, lanePieceLength); 
21P = [xp, yp, phi];
22D = 10000*ones(numel(x_measure), 4); 
23
24for s=1:numel(laneModel)
25   for m=1:numel(x_measure)            
26        dist_point = sqrt((P(s, 1)-(x_measure(m)))^2 + (P(s, 2)-(y_measure(m)))^2);
27        if dist_point < D(m, 3)
28            D(m, 1) = s;
29            D(m, 2) = 0;
30            D(m, 3) = dist_point;     
31            D(m, 4) = -sign(laneModel(min([numel(laneModel), s+1])));  % positive Krümmung -> vorzeichen von d negativ (und anders rum)     
32        end            
33        if s > 1
34            [dist_line, lambda, sgn] = d_line_point([P(s-1, 1), P(s-1, 2)], [P(s, 1), P(s, 2)], [x_measure(m), y_measure(m)]);
35            if dist_line < D(m, 3) && lambda > 0 && lambda < 1
36                D(m, 1) = s-1;
37                D(m, 2) = lambda;
38                D(m, 3) = dist_line; 
39                D(m, 4) = sgn;  % vorzeichen von d
40            end
41        end         
42    end 
43end
44
45y_measure = sign(mean(D(:,4)))*mean(D(:,3));
46x_measure = (mean(D(:,1) + D(:,2))-1)*lanePieceLength;
47
48% initialize the state vector on first object detection
49if init == 1
50    r = [x_measure; 0; y_measure; 0];
51    return
52end
53
54
55%% state space model matrices
56T = 0.01;
57A =    [1, T, 0, 0;
58        0, 1, 0, 0;
59        0, 0, 1, T;
60        0, 0, 0, 1];    
61B = [-1; 0; 0; 0];    
62H = [1, 0, 0, 0;
63     0, 0, 1, 0];
64
65% covariance of the state transition
66d = Q^2;
67QQ = [  d,     d*T,   0,     0;
68        d*T,   d*T^2, 0,     0;
69        0,     0,     d,     d*T;
70        0,     0,     d*T,   d*T^2];
71
72%% Kalman filter 
73
74% prediction
75r = A*r + B*delta_x;
76Pk = A*Pk*A' + QQ;
77
78% kalman gain
79S = H*Pk*H' + [Rx^2, 0; 0 Ry^2];
80K = Pk*H'/S;
81
82% update
83r = r + K*([x_measure; y_measure] - H*r);
84Pk = Pk - K*S*K';
85
